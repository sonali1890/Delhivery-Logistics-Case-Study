def split_name(x):
    try:
        parts = x.split('(')
        city_place = parts[0].strip()
        state = parts[1].replace(')','').strip()
        return city_place, state
    except:
        return np.nan, np.nan

In [18]:
num_cols = trip_df.select_dtypes(include=np.number).columns

for col in num_cols:
    min_val = trip_df[col].min()
    max_val = trip_df[col].max()
    
    if max_val != min_val:   # to avoid division by zero
        trip_df[col] = (trip_df[col] - min_val) / (max_val - min_val)

Min-Max normalization was implemented manually using the formula (x − min) / (max − min). This scaled all numerical features between 0 and 1, which is more interpretable for logistics data such as time and distance.
